	WA_ALV      TYPE TY_USERSAP,
        WA_FIELDCAT TYPE SLIS_FIELDCAT_ALV.

	TI_ALV     TYPE TABLE OF TY_USERSAP,
        TI_FIELDCAT TYPE SLIS_T_FIELDCAT_ALV.

REFRESH TI_FIELDCAT. "Não necessário repetir

  WA_FIELDCAT-COL_POS = 1. "Posição da coluna

  wa_fieldcat-seltext_L = 'CHAVE DO USUÁRIO'. "Nome da coluna
